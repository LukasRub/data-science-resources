# Getting price level rates
url = f'https://api.worldbank.org/v2/country/{france_iso3_code}/indicator/{price_level_indicator}'
_year_range='1999:2019'
url_params = f'?date={_year_range}&format=json'

print('Price level data source: ', url+url_params)

response = https.request('GET', url+url_params)
response_json = json.loads(response.data)

# print(json.dumps(response_json, indent=2, sort_keys=True))
indicator_explanation = response_json[1][0]['indicator']['value']

france_price_level_df = pd.DataFrame(response_json[1], columns=['date', 'value']).sort_values('date')
france_price_level_df.rename(columns=dict(value='price_level'), inplace=True)
france_price_level_df['price_level_explanation'] = indicator_explanation
france_price_level_df.set_index('date', drop=True, inplace=True)

# Calculating price level growth rate
france_price_level_df['price_level_growth_rate'] = np.nan

for year in france_price_level_df.index[1:]:
    year_ix = france_price_level_df.index.get_loc(year)
    prev_year = france_price_level_df.index[year_ix-1]
    france_price_level_df.loc[year, 'price_level_growth_rate'] = \
        france_price_level_df.loc[year, 'price_level'] / france_price_level_df.loc[prev_year, 'price_level']
    
france_price_level_df['price_level_growth_rate_explanation'] = \
    france_price_level_df['price_level_explanation'] + ' (current year / previous year)'

# Re-ordering columns
cols = france_price_level_df.columns
france_price_level_df = france_price_level_df[[cols[0], cols[2], cols[1], cols[-1]]]

print(france_price_level_df.iloc[0,3])

# Calculating GDP growth rate

    year_ix = france_gdp_df.index.get_loc(year)
    prev_year = france_gdp_df.index[year_ix-1]
    france_gdp_df.loc[year, 'gdp_growth_rate'] =  \
        france_gdp_df.loc[year, 'gdp'] / france_gdp_df.loc[prev_year, 'gdp']

    france_gdp_df['gdp_explanation'] + ' (current year / previous year)'

# Re-ordering columns
cols = france_gdp_df.columns
